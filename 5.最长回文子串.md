## 暴力解法

class Solution:
    def longestPalindrome(self, s):

        size = len(s)
        #特判
        if size < 2:
            return s
        
        #记录最长回文子串长度
        max_len = 1
        res = s[0]

        #枚举所有长度大于等于2的子串
        for i in range(size-1):
            for j in range(i+1, size):
                if j-i+1 > max_len and self.valid(s, i, j):
                    max_len = j-i+1
                    res = s[i:j+1]
        return res

    def valis(self, s, left, right):
        while left < right:
            if s[left] != s[right]:
                return False
            left += 1
            right -= 1
        return True

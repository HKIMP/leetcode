```python
class Solution:
    def maxArea(self, height: List[int]) -> int:
        max_area = 0
        i = 0
        j = len(height) -1
        while i<j:
            
            temp = (j - i)*min(height[i], height[j])

            if temp > max_area:
                max_area = temp
            
            if height[i] < height[j]:
                i += 1
            else:
                j -= 1
        return max_area
```

## 使用辅助栈

时间复杂度：O(1)
空间复杂度：O(n)

stack 与 min_stack同步

        self.min_stack = [math.inf]

不同步

```python
class MinStack:
    def __init__(self):
        
        self.stack = []
        self.min_stack = []

    def push(self, x: int) -> None:
        self.stack.append(x)
        #先判断 合不合适放入栈
        if self.min_stack == [] or x <= self.min_stack[-1]:
            self.min_stack.append(x)
        
    def pop(self) -> None:
        top = self.stack.pop()
        if self.min_stack and top == self.min_stack[-1]:
            self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]            

    def getMin(self) -> int:
        return self.min_stack[-1]
```
## 不使用辅助栈

```python
class MinStack():

    def __init__(self):
        self.stack = []

    def push(self, x):
        if not self.stack:
            self.stack.append((x, x))
        else:
            self.stack.append((x, min(x, self.stack[-1][1])))

    def pop(self):
        self.stack.pop()
    
    def top(self):
        return self.stack[-1][0]

    def getMin(self):
        return self.stack[-1][1]        
```

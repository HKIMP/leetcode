1. left<right
   
```python
class Solution:
    def searchInsert(self, nums:List[int], target: int) -> int:

        n = len(nums)

        if n == 0:
            return 0
        
        if nums[-1] < target:
            return n
        
        left = 0
        right = n-1

        while left < right:
            mid = (left + right) // 2
            if nums[mid] < target:
                left = left + 1
            else:
                right = mid
        return left
```
